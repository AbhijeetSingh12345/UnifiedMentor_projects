    <script>
        // JavaScript Game Logic for Tic-Tac-Toe

        class TicTacToeGame {
            constructor() {
                // Initialize game state
                this.board = Array(9).fill(''); // Empty board represented as array
                this.currentPlayer = 'X'; // X always starts first
                this.gameActive = true; // Track if game is still active
                this.moveCount = 0; // Track number of moves made
                
                // Get DOM elements
                this.cells = document.querySelectorAll('.cell');
                this.currentPlayerText = document.getElementById('current-player-text');
                this.playerIndicator = document.getElementById('player-indicator');
                this.gameStatus = document.getElementById('game-status');
                this.resetButton = document.getElementById('reset-button');
                
                // Define winning combinations (indices of board array)
                this.winningCombinations = [
                    [0, 1, 2], // Top row
                    [3, 4, 5], // Middle row
                    [6, 7, 8], // Bottom row
                    [0, 3, 6], // Left column
                    [1, 4, 7], // Middle column
                    [2, 5, 8], // Right column
                    [0, 4, 8], // Diagonal top-left to bottom-right
                    [2, 4, 6]  // Diagonal top-right to bottom-left
                ];
                
                // Initialize the game
                this.initializeGame();
            }

            // Set up event listeners and initial display
            initializeGame() {
                // Add click event listeners to all cells
                this.cells.forEach(cell => {
                    cell.addEventListener('click', this.handleCellClick.bind(this));
                });
                
                // Add reset button event listener
                this.resetButton.addEventListener('click', this.resetGame.bind(this));
                
                // Update initial display
                this.updateDisplay();
            }

            // Handle cell click events
            handleCellClick(event) {
                const clickedCell = event.target;
                const cellIndex = parseInt(clickedCell.getAttribute('data-index'));
                
                // Check if move is valid
                if (!this.isValidMove(cellIndex)) {
                    return;
                }
                
                // Make the move
                this.makeMove(cellIndex, clickedCell);
                
                // Check for game end conditions
                this.checkGameEnd();
                
                // Switch players if game is still active
                if (this.gameActive) {
                    this.switchPlayer();
                }
                
                // Update display
                this.updateDisplay();
            }

            // Check if a move is valid
            isValidMove(cellIndex) {
                return this.gameActive && this.board[cellIndex] === '';
            }

            // Execute a move
            makeMove(cellIndex, cellElement) {
                // Update board state
                this.board[cellIndex] = this.currentPlayer;
                
                // Update cell display
                cellElement.textContent = this.currentPlayer;
                cellElement.classList.add(this.currentPlayer.toLowerCase());
                cellElement.classList.add('disabled');
                
                // Increment move count
                this.moveCount++;
            }

            // Switch to the other player
            switchPlayer() {
                this.currentPlayer = this.currentPlayer === 'X' ? 'O' : 'X';
            }

            // Check if the game has ended (win or tie)
            checkGameEnd() {
                // Check for winning condition
                const winningCombination = this.checkWin();
                if (winningCombination) {
                    this.handleWin(winningCombination);
                    return;
                }
                
                // Check for tie condition
                if (this.checkTie()) {
                    this.handleTie();
                    return;
                }
            }

            // Check if current player has won
            checkWin() {
                for (let combination of this.winningCombinations) {
                    const [a, b, c] = combination;
                    if (this.board[a] && 
                        this.board[a] === this.board[b] && 
                        this.board[a] === this.board[c]) {
                        return combination;
                    }
                }
                return null;
            }

            // Check if the game is a tie
            checkTie() {
                return this.moveCount === 9;
            }

            // Handle win condition
            handleWin(winningCombination) {
                this.gameActive = false;
                
                // Highlight winning cells
                winningCombination.forEach(index => {
                    this.cells[index].classList.add('winning-cell');
                });
                
                // Disable all cells
                this.disableAllCells();
                
                // Update game status
                this.gameStatus.textContent = `Player ${this.currentPlayer} Wins! ðŸŽ‰`;
                this.gameStatus.className = 'game-status status-winner';
            }

            // Handle tie condition
            handleTie() {
                this.gameActive = false;
                
                // Disable all cells
                this.disableAllCells();
                
                // Update game status
                this.gameStatus.textContent = "It's a Tie! ðŸ¤";
                this.gameStatus.className = 'game-status status-tie';
            }

            // Disable all game cells
            disableAllCells() {
                this.cells.forEach(cell => {
                    cell.classList.add('disabled');
                });
            }

            // Update the display elements
            updateDisplay() {
                // Update current player display
                if (this.gameActive) {
                    this.currentPlayerText.textContent = this.currentPlayer;
                    this.playerIndicator.className = `player-indicator player-${this.currentPlayer.toLowerCase()}`;
                    this.gameStatus.textContent = 'Game in Progress';
                    this.gameStatus.className = 'game-status status-ongoing';
                }
            }

            // Reset the game to initial state
            resetGame() {
                // Reset game state
                this.board = Array(9).fill('');
                this.currentPlayer = 'X';
                this.gameActive = true;
                this.moveCount = 0;
                
                // Reset all cells
                this.cells.forEach(cell => {
                    cell.textContent = '';
                    cell.className = 'cell'; // Remove all additional classes
                });
                
                // Update display
                this.updateDisplay();
            }
        }

        // Initialize the game when the page loads
        document.addEventListener('DOMContentLoaded', function() {
            const game = new TicTacToeGame();
        });
    </script>
